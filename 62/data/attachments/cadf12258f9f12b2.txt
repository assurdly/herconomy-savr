try {
    const jsonData = pm.response.json();

    pm.test("Status code is 400", function () {
        pm.response.to.have.status(400);
    });

    pm.test("Response body has property 'detail'", function () {
        pm.expect(jsonData).to.have.property("detail");
        pm.expect(jsonData.detail).to.be.an("string");
    });
 pm.test("Status code response is Bad Request ", function () {
        pm.response.to.have.status('Bad Request');
    });
    pm.test('Verify the error message in the response', () => {
        pm.expect(jsonData.detail).eql("Password not the same")
    })
} catch (errors) {
    console.error(errors)
}
